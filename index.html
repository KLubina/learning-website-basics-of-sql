SQL Joins Visualisiert

INNER JOIN

SELECT * FROM Tabelle1 INNER JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id;

LEFT JOIN

SELECT * FROM Tabelle1 LEFT JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id;

RIGHT JOIN

SELECT * FROM Tabelle1 RIGHT JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id;

FULL OUTER JOIN

SELECT * FROM Tabelle1 FULL OUTER JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id;

LEFT JOIN EXCLUDING INNER JOIN

SELECT * FROM Tabelle1 LEFT JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id WHERE Tabelle2.id IS NULL;

RIGHT JOIN EXCLUDING INNER JOIN

SELECT * FROM Tabelle1 RIGHT JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id WHERE Tabelle1.id IS NULL;

FULL OUTER JOIN EXCLUDING INNER JOIN

SELECT * FROM Tabelle1 FULL OUTER JOIN Tabelle2 ON Tabelle1.id = Tabelle2.id WHERE Tabelle1.id IS NULL OR Tabelle2.id IS NULL;

SQL Kommandos

DDL (Data Definition Language)

  • CREATE CREATE TABLE table_name (column1 datatype, column2 datatype, ...);
  • DROP DROP TABLE table_name;
  • ALTER ALTER TABLE table_name ADD column_name datatype;
  • TRUNCATE TRUNCATE TABLE table_name;

DML (Data Manipulation Language)

  • INSERT INSERT INTO table_name (column1, column2, ...) VALUES (value1, value2, ...);
  • UPDATE UPDATE table_name SET column1 = value1, column2 = value2, ... WHERE condition;
  • DELETE DELETE FROM table_name WHERE condition;
  • CALL CALL procedure_name (parameter1, parameter2, ...);
  • EXPLAIN CALL EXPLAIN CALL procedure_name (parameter1, parameter2, ...);
  • LOCK LOCK TABLE table_name IN {SHARED|EXCLUSIVE} MODE;

TCL (Transaction Control Language)

  • COMMIT COMMIT;
  • SAVEPOINT SAVEPOINT savepoint_name;
  • ROLLBACK ROLLBACK TO savepoint_name;
  • SET TRANSACTION SET TRANSACTION {READ ONLY|READ WRITE};
  • SET CONSTRAINT SET CONSTRAINTS constraint_name {DEFERRED|IMMEDIATE};

DQL (Data Query Language)

  • SELECT SELECT column1, column2, ... FROM table_name WHERE condition;

DCL (Data Control Language)

  • GRANT GRANT privilege_name ON object TO {user|role};
  • REVOKE REVOKE privilege_name ON object FROM {user|role};
